# Atheer Simulation Algorithm v2.1
# Parameters derived from [Ookla] and [Internet Society] reports

import numpy as np

class NetworkSimulation:
    def __init__(self, network_stress_level):
        # Public Internet Configuration (Yemen Baseline)
        self.public_avg_latency = 5.0  # seconds (Average delay)
        self.public_jitter = 2.5       # High instability
        self.packet_loss_prob = network_stress_level / 100.0 

        # Atheer Private APN Configuration (Isolated Tunnel)
        self.atheer_avg_latency = 0.8  # seconds (Fast)
        self.atheer_jitter = 0.1       # Very Stable

    def run_public_transaction(self):
        """Simulates public internet congestion & retries"""
        latency = np.random.normal(self.public_avg_latency, self.public_jitter)
        
        # Simulating Packet Loss Penalty (TCP Retransmission)
        if np.random.random() < self.packet_loss_prob:
            latency += 5.0 + np.random.normal(2.0, 1.0) # Penalty
            
        return max(2.0, latency) # Minimum round-trip time

    def run_atheer_transaction(self):
        """Simulates dedicated tunnel via Cellular Core"""
        # HCE Processing (Device) + Tunnel Time (Network)
        latency = np.random.normal(self.atheer_avg_latency, self.atheer_jitter)
        return max(0.5, latency)
